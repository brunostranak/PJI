DROP DATABASE PJI;
CREATE DATABASE PJI;
USE PJI;

CREATE TABLE usuarios(
idUser INT AUTO_INCREMENT,
nomeUser VARCHAR(60),
dtNasc DATE,
genero VARCHAR(20),
curso VARCHAR(60),
email VARCHAR(60),
senha VARCHAR(20),
imagem VARCHAR(20),
telefone VARCHAR(20),
bio VARCHAR(450),
PRIMARY KEY(idUser)
) engine = innodb;

CREATE TABLE livros(
idLivro INT AUTO_INCREMENT,
idUser INT,
nomeLivro VARCHAR(60),
autor VARCHAR(60),
editora VARCHAR(60),
status VARCHAR(60),
resumo VARCHAR(300),
imagem VARCHAR(20),
qtdFeed INT(2),
PRIMARY KEY (idLivro),
FOREIGN KEY (idUser) REFERENCES usuarios (idUser)
) engine = innodb;


CREATE TABLE emprestimos(
idEmprestimo INT AUTO_INCREMENT,
idUser INT,
idLivro INT,
idEmprestante INT,
dtInicio DATE,
dtFim DATE,


PRIMARY KEY (idEmprestimo),
FOREIGN KEY (idUser) REFERENCES usuarios (idUser),
FOREIGN KEY (idLivro) REFERENCES livros (idLivro)
) engine = innodb;


CREATE TABLE feedbacks(
idFeedback INT AUTO_INCREMENT,
idUser INT,
idLivro INT,
feedback VARCHAR(450),



PRIMARY KEY (idFeedback),
FOREIGN KEY (idUser) REFERENCES usuarios (idUser),
FOREIGN KEY (idLivro) REFERENCES livros (idLivro)
) engine = innodb;


CREATE TABLE notific(
idNotific INT AUTO_INCREMENT,
idUser INT,
idEmprestimo INT,
texto VARCHAR(450),
dtInicio DATE,

PRIMARY KEY (idNotific),
FOREIGN KEY (idUser) REFERENCES usuarios (idUser),
FOREIGN KEY (idEmprestimo) REFERENCES emprestimos (idEmprestimo)
) engine = innodb;

CREATE TABLE notific_temp(
idNotific INT AUTO_INCREMENT,
idUser INT,
idEmprestimo INT,
texto VARCHAR(450),
dtInicio DATE,

PRIMARY KEY (idNotific),
FOREIGN KEY (idUser) REFERENCES usuarios (idUser),
FOREIGN KEY (idEmprestimo) REFERENCES emprestimos (idEmprestimo)
) engine = innodb;
